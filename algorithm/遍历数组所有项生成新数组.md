## 遍历数组所有项生成新数组
```js
var arr = [[1, 2, 3], [5], [6, 78]];
var brr = [];

function channge() {
    while (arr.length > 0) {
        var i = 0;
        for (var j = 0; j < arr[i].length; j++) {
            brr.push(arr[i][j]);
        }
        arr.shift();
        i++;
    }
    console.log(arr);//最终打印为[]
    console.log(brr);//最终打印为[ 1, 2, 3, 5, 6, 78 ]
}

channge();
```
